  console.log(a); // a = 1

      # 比較 let 與 var

      let 的作用域只存在大括號內

      // a is not defined

// example 1
{
  let a = 1;
}
console.log(a); // a is not defined

//example 2
function fn() {
  let a = 1;
}
console.log(a); // a is not defined

      let 雖然有類似於 var 的 hoisting 行為，但是不允許提前取值。 在這個宣告與取值之間的區塊有個名詞叫做，暫時性死區（ TDZ ） 因此，宣告時應盡量在開頭宣告變數。（ 我也不懂為什麼有人不在開頭宣告變數

      function fn(a) {
  console.log(a) // a = 1

  // example 1, 可以跑
  var a = 2;
  console.log(a); // a = 2

  // example 2, 會報錯
  let a = 2;
  console.log(a);
}
fn(1);
