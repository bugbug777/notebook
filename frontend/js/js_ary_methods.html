                        # 使用 map 回傳、組合字串

let str = '';
people.forEach(item => {
  str += `<li>${ item.order }, ${item.price}</li>`;
})

// Example 2, map
const str = people.map(item => `<li>${ item.order }, ${item.price}</li>`).join('');

                        # 使用 sort 進行排序，會回傳排序後的新陣列

                        // Syntax
ary.sort((a, b) => {
  return a - b; // 透過正負數判斷，進行排序
})

// Example
const newAry = ary.sort((a, b) => a.price - b.price);
